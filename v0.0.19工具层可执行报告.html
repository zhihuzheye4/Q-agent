四、9 个工具实现规范

4.1 file_read

@tool(
    name="file_read",
    desc=(
        "读取文本文件内容并返回字符串。"
        "何时用：需要查看源代码、配置文件、文档内容时。"
        "何时不用：读取二进制文件、超过 1MB 的大文件、敏感凭证文件。"
        "参数约束：path 绝对路径或相对 cwd；命中敏感清单返回 PermissionError。"
        "返回格式：文件全文（超 2000 字符触发预算降级）。"
    ),
    version="1.0.0",
    timeout=15.0,
    permission_level="read_only",
)
def file_read(path: str, encoding: str = "utf-8") -> str:
    check_path(path)
    check_sensitive(path)  # 命中 .env/.pem/id_rsa/credentials.json 等拦截
    p = Path(path)
    if not p.exists():
        return _error_json("FileNotFound", f"文件不存在: {path}")
    if p.stat().st_size > 1_000_000:
        return _error_json("FileTooLarge", f"文件超过 1MB: {p.stat().st_size} bytes")
    return p.read_text(encoding=encoding)

错误回喂 JSON 格式（所有工具统一）：

{"error": "FileNotFound", "message": "文件不存在: /foo/bar.py", "recoverable": true}

recoverable: true = LLM 可自愈重试；recoverable: false = PermissionError 硬拦截零重试。

4.2 file_write

@tool(
    name="file_write",
    desc=(
        "将字符串内容写入文件（覆盖模式）。"
        "何时用：创建新文件、修改源代码、生成配置文件。"
        "何时不用：追加内容、写二进制。"
        "参数约束：path + content；命中敏感文件或内容含 PRIVATE KEY/AKIA 模式拒绝。"
        "返回格式：写入字节数 + 落盘备份路径。"
    ),
    version="1.0.0",
    timeout=10.0,
    permission_level="write",
)
def file_write(path: str, content: str, encoding: str = "utf-8") -> str:
    check_path(path)
    check_sensitive(path)
    _sniff_secret_content(content)  # 命中 PRIVATE KEY/AKIA[0-9A-Z]{16} 等拒绝
    backup = snapshot_before_write(path)  # 写前快照
    Path(path).write_text(content, encoding=encoding)
    return f"写入 {len(content.encode(encoding))} bytes; backup={backup}"

撤销策略：写前快照备份到 ~/.q-agent/snapshots/{call_id}/，undo 栈容量 20。

4.7 exec_shell

@tool(
    name="exec_shell",
    desc=(
        "执行 shell 命令（非交互、捕获 stdout/stderr）。"
        "何时用：运行测试、跑构建、查 git 状态、装依赖（在用户授权后）。"
        "何时不用：长时间运行的服务进程、交互输入命令、批量删除（危险）。"
        "参数约束：cmd 为 list[str] 禁 shell=True；long_running=True 时超时 600s。"
        "返回格式：returncode + stdout（前 2000 字符）+ stderr（前 500 字符）。"
    ),
    version="1.0.0",
    timeout=120.0,
    long_running=False,
    permission_level="destructive",
    needs_confirmation=True,  # 每次都问，destructive 永不缓存
)
def exec_shell(cmd: list[str], cwd: str | None = None, long_running: bool = False) -> str:
    from q_agent.tools.executor import execute_shell
    timeout = 600.0 if long_running else 120.0
    cp = execute_shell(cmd, cwd=cwd, timeout=timeout)
    out = cp.stdout[:2000]
    err = cp.stderr[:500]
    return f"returncode={cp.returncode}\nstdout:\n{out}\nstderr:\n{err}"

4.8 web_get

@tool(
    name="web_get",
    desc=(
        "对白名单域名发起 HTTP GET，返回原始响应体（前 2000 字符）。"
        "何时用：拉取 API JSON、查询文档站、检查连通性。"
        "何时不用：抓取 HTML 转 markdown（用 web_fetch）、上传数据、访问私网。"
        "参数约束：url 必须 HTTPS 且域名在白名单；私网段拒绝（SSRF 防护）。"
        "返回格式：status_code + body 前 2000 字符。"
    ),
    version="1.0.0",
    timeout=30.0,
    permission_level="read_only",
)
def web_get(url: str) -> str:
    check_url(url)   # HTTPS + 域名白名单
    check_ssrf(url)  # 私网段拒绝
    with urllib.request.urlopen(url, timeout=30) as r:
        body = r.read(1_000_000).decode("utf-8", errors="ignore")
    return f"status={r.status}\nbody:\n{body[:2000]}"

7.3 ContextManager L1 与 tool_calls 联动

# 会话总预算 24000 字符超限时，按三档分级降级：
# L1 占位：tool_call 输出替换为 "[tool:{tool_name} output 已落盘 {path}]"
# L2 截断：保留前 200 + 后 200 字符
# L3 摘要：调 LLM 生成 100 字摘要替换

def compact_tool_outputs(self, session_id: str) -> None:
    """会话总预算超限时，按 L1→L2→L3 逐档降级 tool_calls 输出。"""
    rows = self.persistence.load_tool_calls(session_id)
    total = sum(len(r["output_text"] or "") for r in rows)
    if total <= 24000: return
    for r in rows:
        if total <= 24000: break
        total -= len(r["output_text"] or "")
        new_text = f"[tool:{r['tool_name']} output 已落盘 {r['output_path']}]"
        self.persistence.update_tool_call_output(r["call_id"], new_text)
        total += len(new_text)
